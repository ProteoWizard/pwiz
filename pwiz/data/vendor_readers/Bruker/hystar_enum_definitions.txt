Definitions for the Type and Unit columns in chromatography-data.sqlite files, per Sven at Bruker 2025/12/11
--------------

Trace-Type:

    /// Default none-value.
    NoneTrace		= 0
 
    /// Trace from MS data
    ChromMS			= 1
 
    /// Trace from UV detector
    ChromUV			= 3
 
    /// Trace of a pressure.
    /// But note that this pressure might be the value of a pressure
    /// gauge that can be positioned anywhere - in that case no further
    /// information (except the description) is available.
    ChromPressure	= 4
 
    /// Trace from solvent mix (percentage) of pump.
    ChromSolventMix	= 5
 
    /// <summary>
    /// Trace of flow.
    /// But note that this flow might be the value of a flow gauge that
    /// can be positioned anywhere - in that case no further information
    /// (except the description) is available.
    ChromFlow		= 6
 
    /// Trace with the temperature.
    /// But note that this temperature might be the value of a
    /// temperature gauge that can be positioned anywhere - in that case
    /// no further information (except the description) is available.
    ChromTemperature = 7
 
    /// Some other type of chromatogramm
    ChromUserDefined = 9999
};
