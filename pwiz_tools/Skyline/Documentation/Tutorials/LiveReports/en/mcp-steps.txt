# Live Reports tutorial (en) -- SkylineMCP / AI Connector steps  [PARTIAL -- see gaps at bottom]
#
# English (en) step list. Verbs: InvokeMenuItem, InvokeContextMenuItem, ClickFormButton,
# ClickToolStripItem, SetFormValue, SetGridText, GetGridText, CloseForm, and the WaitForForm /
# WaitForNoForm sync primitives. Each localized tutorial needs its own; {DATA} is not localized.
#
# {DATA} is the LiveReports folder that contains Rat_plasma.sky and SampleInfo.txt
# (https://skyline.ms/tutorials/LiveReports.zip).
#
# STATUS: validated live through the Document Grid section. The tutorial then reaches the "Define
# Annotation" dialog, which is blocked on two remaining verb gaps (plain DataGridView cells and a
# CheckedListBox item -- see the gap list). Steps below are validated unless marked BLOCKED.

# --- Open the starting document ---
# (If a prior document has unsaved changes, File > Open first shows a "Save changes?" prompt
# (MultiButtonMsgDlg) -- answer No to discard. Validated.)
InvokeMenuItem  "File > Open"
WaitForForm     "FileDialog:Open"
SetFormValue    "FileDialog:Open" "FileName" {DATA}\Rat_plasma.sky
ClickFormButton "FileDialog:Open" "Open"
WaitForNoForm   "FileDialog:Open"
# Document: 48 proteins, 125 peptides, 721 transitions, 42 replicates.

# --- Enable Audit Logging ---
InvokeMenuItem  "View > Live Reports > Audit Log"
WaitForForm     "AuditLogForm:Audit Log: All Info"
# "Enable audit logging" is a CheckBox with AutoCheck=false that acts on its Click handler, so it
# must be CLICKED (ClickFormButton), not value-set. Validated.
ClickFormButton "AuditLogForm:Audit Log: All Info" "Enable audit logging"
# Close the Audit Log window. CloseForm now closes a docked/floating tool window. Validated.
CloseForm       "AuditLogForm:Audit Log: All Info"

# --- Showing the Document Grid ---
# Note: the Document Grid opens on whichever view the saved document last used (here "Peptides"),
# so the form title -- and thus the form id -- is "Document Grid: <ViewName>". Read it back from
# GetOpenForms rather than assuming "Proteins".
InvokeMenuItem  "View > Live Reports > Document Grid"
WaitForForm     "DocumentGridForm:Document Grid: *"
# The "Reports" dropdown items are built on demand; ClickToolStripItem opens the dropdown first.
# Validated (Proteins, Replicates).
ClickToolStripItem "DocumentGridForm:Document Grid: Peptides" "Reports > Proteins"
# GetGridText returns the whole grid (headers + every row) as TSV. Validated (48-protein report).
GetGridText     "DocumentGridForm:Document Grid: Proteins" null
ClickToolStripItem "DocumentGridForm:Document Grid: Proteins" "Reports > Replicates"

# --- Defining some Replicate Annotations ---
# SubjectID (a plain Text annotation) is now fully driveable end-to-end. Validated.
InvokeMenuItem  "Settings > Document Settings"
WaitForForm     "DocumentSettingsDlg:Document Settings"
ClickFormButton "DocumentSettingsDlg:Document Settings" "Add"
WaitForForm     "DefineAnnotationDlg:Define Annotation"
# NOTE (gap 10): use the textbox control NAME "tbxName", NOT the label "Name". SetFormValue matches
#   the "Name:" LABEL by its visible text and would set the label instead of the input. See gap (10).
SetFormValue    "DefineAnnotationDlg:Define Annotation" "tbxName" "SubjectID"
# Type defaults to "Text". Check "Replicates" in the Applies-to CheckedListBox (ClickFormButton now
# toggles a checked-list item). Validated.
ClickFormButton "DefineAnnotationDlg:Define Annotation" "Replicates"
ClickFormButton "DefineAnnotationDlg:Define Annotation" "OK"
ClickFormButton "DocumentSettingsDlg:Document Settings" "OK"

# Fill the SubjectID column in the Replicates report. This grid IS a DataboundGridControl, so
# SetGridText pastes the whole 42-row column at once. Validated (read back with GetGridText).
InvokeMenuItem  "View > Live Reports > Document Grid"
ClickToolStripItem "DocumentGridForm:Document Grid: *" "Reports > Replicates"
SetGridText     "DocumentGridForm:Document Grid: Replicates" "" 3 0 "102\n102\n102\n...(42 rows)"

# The "Cohort" Value-List annotation is now driveable: its "Values" control is a multi-line TextBox
# (not a grid), so after SetFormValue "Type" "Value List", set the values with
#   SetFormValue "DefineAnnotationDlg:Define Annotation" "Values" "Healthy\nDiseased"
# (the "Values" label resolves to the textbox via gap (10)).
