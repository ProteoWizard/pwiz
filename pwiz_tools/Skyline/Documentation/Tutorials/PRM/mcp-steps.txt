# PRM tutorial -- SkylineMCP / AI Connector steps
#
# Each line is one operation: a verb name, then its arguments.
# Lines starting with '#' and blank lines are ignored.
#
# Argument format:
#   - formId and controlId arguments are wrapped in double quotes.
#   - The VALUE argument of SetFormValue is the rest of the line after the controlId
#     (taken verbatim, so it may itself contain spaces and double quotes -- needed for
#     multi-file selection, see below).
#
# Verbs that map directly to IJsonToolService methods:
#   InvokeMenuItem  "<menu path>"                  -> skyline_invoke_menu_item
#   ClickFormButton "<formId>" "<button>"          -> skyline_click_form_button
#   SetFormValue    "<formId>" "<controlId>" <value>  -> skyline_set_form_value
#
# Harness synchronization primitives (poll skyline_get_open_forms; NOT MCP methods):
#   WaitForForm     "<formId>"   -- wait until that form/dialog is listed by GetOpenForms
#   WaitForNoForm   "<formId>"   -- wait until it is gone
# Required because InvokeMenuItem / ClickFormButton return immediately (the dialog-watch
# makes them return as soon as a dialog appears), so the resulting form is usually not
# open yet on the next line.
#
# formIds are the stable "TypeName:Title" identifiers from GetOpenForms; they are
# deterministic for this tutorial, so no run-time inspection/branching is needed --
# GetOpenForms is used only for the Wait* steps. (The titles below are the English UI
# strings; a localized run needs the localized titles.)
#
# {DATA} is replaced by the test with the PRM tutorial data directory (the folder that
# contains BSA_Protea_label_free_meth3.sky and the 'search' subfolder).
#
# Validation: every step below was executed against Skyline-daily over the pipe, through
# adding both search files. The tutorial continues past this point (Next -> build spectral
# library -> match modifications -> ... -> Finish -> import raw data); not captured yet.

# --- Open the starting document ---
InvokeMenuItem  "File > Open"
WaitForForm     "FileDialog:Open"
SetFormValue    "FileDialog:Open" "FileName" {DATA}\BSA_Protea_label_free_meth3.sky
ClickFormButton "FileDialog:Open" "Open"
WaitForNoForm   "FileDialog:Open"

# --- File > Import > Peptide Search (document is saved, so no save prompt) ---
InvokeMenuItem  "File > Import > Peptide Search"
WaitForForm     "ImportPeptideSearchDlg:Import Peptide Search"

# --- Add both search-result files at once ---
# Multi-select on the common (classic) Open dialog: navigate INTO the folder first, then
# set the file name to the QUOTED BARE filenames separated by spaces. (Setting quoted
# FULL paths is rejected with "Operation cannot be performed"; bare names in the current
# folder work.)
ClickFormButton "ImportPeptideSearchDlg:Import Peptide Search" "Add Files"
WaitForForm     "FileDialog:Add Input Files"
SetFormValue    "FileDialog:Add Input Files" "FileName" {DATA}\search
ClickFormButton "FileDialog:Add Input Files" "Open"
SetFormValue    "FileDialog:Add Input Files" "FileName" "klc_20100329v_Protea_Peptide_Curve_20fmol_uL_tech1.perc.xml" "klc_20100329v_Protea_Peptide_Curve_80fmol_uL_tech1.perc.xml"
ClickFormButton "FileDialog:Add Input Files" "Open"
WaitForNoForm   "FileDialog:Add Input Files"

# Both .perc.xml files are now in the wizard's spectral-library build grid.
SetFormValue    "ImportPeptideSearchDlg:Import Peptide Search" "cbFilterForDocumentPeptides" true
SetFormValue    "ImportPeptideSearchDlg:Import Peptide Search" "gridSearchFiles" 0.01
ClickFormButton "ImportPeptideSearchDlg:Import Peptide Search" "Next"

# Wait for library build progress (it's a modal dialog but we can wait for its disappearance)
# WaitForForm     "LongWaitDlg:Building Library" # Optional, WaitForNoForm usually enough if we know it was there
# WaitForNoForm   "LongWaitDlg:Building Library"

# Second page: Extract Chromatograms (form title usually doesn't change, but content does)
# Skyline should have automatically found the raw files in {DATA}
ClickFormButton "ImportPeptideSearchDlg:Import Peptide Search" "Next"

# Remove Prefix/Suffix dialog
WaitForForm     "ImportResultsNameDlg:Import Results"
ClickFormButton "ImportResultsNameDlg:Import Results" "OK"
WaitForNoForm   "ImportResultsNameDlg:Import Results"

# Transition Settings page
ClickFormButton "ImportPeptideSearchDlg:Import Peptide Search" "Next"

# Full-Scan Settings page
ClickFormButton "ImportPeptideSearchDlg:Import Peptide Search" "Finish"

# File import starts. We can wait for the "Loading chromatograms" form to appear and then disappear.
WaitForForm     "AllChromatogramsGraph:Loading chromatograms"
WaitForNoForm   "AllChromatogramsGraph:Loading chromatograms"

# --- Reviewing the Data ---
# Select the first peptide to view its chromatograms
SetSelection    /MoleculeGroup[1]/Molecule[1]

#    'FileNameControlHost' has no settable element; buttons are DirectUI panes). They are
#    neither listed by GetOpenForms nor driveable yet. This tutorial avoids them by
#    opening an already-saved document; saving would need RunCommand --out instead.
